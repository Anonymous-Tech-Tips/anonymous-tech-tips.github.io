  <script>
    const grid = document.getElementById("grid");
    const search = document.getElementById("search");
    const countEl = document.getElementById("count");
    let games = [];

    fetch("games.json?v=14", { cache: "no-store" })
      .then(r => r.json())
      .then(data => {
        games = data;
        render(games);
      });

    function render(list) {
      grid.innerHTML = "";
      const playable = list.filter(g => !g.broken);
      list.forEach(g => {
        const a = document.createElement("a");
        a.className = "card" + (g.broken ? " broken" : "");
        if (!g.broken) a.href = `games/${g.slug}/${g.entry}`;
        else a.href = "#";
        a.dataset.name = (g.name + " " + g.slug).toLowerCase();
        if (g.broken) a.title = "Missing game files in archive — cannot run on static hosting";

        const icon = document.createElement("div");
        icon.className = "icon";
        icon.style.background = `linear-gradient(135deg, hsl(${g.hue} 55% 45%), hsl(${(g.hue + 40) % 360} 60% 35%))`;
        icon.textContent = g.slug.slice(0, 2);

        const label = document.createElement("span");
        label.className = "label";
        label.textContent = g.name;

        a.append(icon, label);
        grid.appendChild(a);
      });
      const brokenN = list.filter(g => g.broken).length;
      countEl.textContent = brokenN
        ? `${playable.length} playable · ${brokenN} unavailable`
        : `${list.length} game${list.length === 1 ? "" : "s"}`;
    }

    search.addEventListener("input", () => {
      const q = search.value.trim().toLowerCase();
      if (!q) return render(games);
      render(games.filter(g =>
        g.name.toLowerCase().includes(q) || g.slug.toLowerCase().includes(q)
      ));
    });
  </script>
